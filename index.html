from reducto import Reducto

client = Reducto(api_key="sk_live_...")

# Parse any document into structured data
result = client.parse.run(
    document_url="https://files.acme.co/10-K.pdf",
    options={"table_mode": "agentic"},
)

print(result.tables[0].to_markdown())
print(result.confidence)  # 0.997
# Parse a document with a single request
